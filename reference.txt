        function escapeHTML(str) {
            if (!str) return "";
            return str.replace(/[&<>"']/g, function(m) {
                switch (m) {
                    case '&': return '&amp;';
                    case '<': return '&lt;';
                    case '>': return '&gt;';
                    case '"': return '&quot;';
                    case "'": return '&#039;';
                    default: return m;
                }
            });
        }

        function setupEventListeners() {
            const searchBox = document.getElementById("searchBox");
            
            searchBox.addEventListener("input", (e) => {
                toggleClearBtn(); 
                debounce(() => showSuggestions(e.target.value), 300)();
            });
            
            // KEYBOARD NAVIGATION LOGIC
            searchBox.addEventListener("keydown", (e) => { 
                const box = document.getElementById("suggestions");
                let items = box.getElementsByClassName("suggestion-item");
                
                if (e.key === "ArrowDown") {
                    currentFocus++;
                    addActive(items);
                    e.preventDefault(); 
                } else if (e.key === "ArrowUp") {
                    currentFocus--;
                    addActive(items);
                    e.preventDefault();
                } else if (e.key === "Enter") {
                    e.preventDefault();
                    if (currentFocus > -1 && box.classList.contains("active")) {
                        if (items[currentFocus]) {
                            items[currentFocus].click();
                        }
                    } else {
                        box.classList.remove("active"); 
                        searchNovel(); 
                    }
                }
            });

            document.addEventListener("click", (e) => { 
                if (!document.querySelector(".search-container").contains(e.target)) { 
                    document.getElementById("suggestions").classList.remove("active"); 
                } 
            });

        function addActive(items) {
            if (!items) return false;
            removeActive(items);
            if (currentFocus >= items.length) currentFocus = 0;
            if (currentFocus < 0) currentFocus = (items.length - 1);
            
            if(items[currentFocus].classList.contains('no-hover')) {
                currentFocus = -1;
                return;
            }
            
            items[currentFocus].classList.add("selected");
            items[currentFocus].scrollIntoView({ behavior: 'smooth', block: 'nearest' });
        }

            if (query) {

        async function showSuggestions(query) {
            const box = document.getElementById("suggestions");
            currentFocus = -1; 

            if (!query) { box.classList.remove("active"); return; }
            const isRestricted = RESTRICTED_KEYWORDS.some(word => query.toLowerCase().includes(word));
            if (isRestricted) { box.classList.remove("active"); return; }
            
            try {
                const res = await fetch(`${CONFIG.apiUrl}?query=${encodeURIComponent(query)}`);
                const result = await res.json();
                
                const suggestionsList = (result.data || []).slice(0, 5);
                
                if (suggestionsList.length === 0) { 
                    box.classList.remove("active");
                    return; 
                }
                
                box.innerHTML = suggestionsList.map(r => {
                    const safeTitle = escapeHTML(r.Titles);
                    const jsSafeTitle = r.Titles.replace(/'/g, "\\'"); 
                    
                    let highlightedTitle = safeTitle;
                    try {
                        const safeQuery = query.replace(/[.*+?^${}()|[\]\\]/g, '\\$&');
                        const regex = new RegExp(`(${safeQuery})`, 'gi');
                        highlightedTitle = safeTitle.replace(regex, '<span style="color: var(--sn-ink); font-weight: 800;">$1</span>');
                    } catch(e) {}

                    return `<div class="suggestion-item" onclick="selectSuggestion('${jsSafeTitle}')">
                                <span class="suggestion-icon">
                                    <svg focusable="false" xmlns="http://www.w3.org/2000/svg" viewBox="0 0 24 24" width="18" height="18" fill="currentColor"><path d="M15.5 14h-.79l-.28-.27A6.471 6.471 0 0 0 16 9.5 6.5 6.5 0 1 0 9.5 16c1.61 0 3.09-.59 4.23-1.57l.27.28v.79l5 4.99L20.49 19l-4.99-5zm-6 0C7.01 14 5 11.99 5 9.5S7.01 5 9.5 5 14 7.01 14 9.5 11.99 14 9.5 14z"></path></svg>
                                </span>
                                <span>${highlightedTitle}</span>
                            </div>`;
                }).join("");
                
                box.classList.add("active");
            } catch(e) {
                box.classList.remove("active");
            }
        }

        function toggleFav(title, link, btn) {
            const idx = state.favorites.findIndex(f => f.Titles === title);

        function renderPagination(totalPages) {
            const div = document.getElementById("pagination");
            div.innerHTML = "";
            if (totalPages <= 1) return;
            const { currentPage } = state;
            const createBtn = (page, text, isNav = false, isDisabled = false) => {
                const btn = document.createElement("button");
                btn.className = `page-btn ${isNav ? 'nav-btn' : ''} ${page === currentPage && !isNav ? 'active' : ''}`;
                btn.textContent = text || page;
                btn.disabled = isDisabled;
                if (!isDisabled) { 
                    btn.onclick = () => { 
                        state.currentPage = page; 
                        if (state.showFavorites) { renderResults(); } 
                        else { fetchData(document.getElementById("searchBox").value.trim()); }
                        document.getElementById("novel-app-wrapper").scrollIntoView({ behavior: 'smooth' }); 
                    }; 
                }
                return btn;
            };
            const createEllipsis = () => { const span = document.createElement("span"); span.className = "pagination-ellipsis"; span.innerHTML = "&hellip;"; return span; };

            div.appendChild(createBtn(currentPage - 1, "Prev", true, currentPage === 1));
            const pages = []; pages.push(1);
            let start = Math.max(2, currentPage - 1); let end = Math.min(totalPages - 1, currentPage + 1);
            if (start > 2) pages.push('...');
            for (let i = start; i <= end; i++) { if (i > 1 && i < totalPages) pages.push(i); }
            if (end < totalPages - 1) pages.push('...');
            if (totalPages > 1) pages.push(totalPages);
            pages.forEach(page => { if (page === '...') div.appendChild(createEllipsis()); else div.appendChild(createBtn(page)); });
            div.appendChild(createBtn(currentPage + 1, "Next", true, currentPage === totalPages));
        }

        async function surpriseMe(isReroll = false) {
            const overlay = document.getElementById("surpriseOverlay");
            const title = document.getElementById("surpriseTitle");
            const dice = document.getElementById("surpriseDice");
            const dlBtn = document.getElementById("surpriseDownload");
            const againBtn = document.querySelector('.surprise-again');
            document.querySelector('.surprise-btn').classList.add('paused');
            
            if(againBtn) againBtn.classList.remove('attention');

            overlay.classList.add("active");
            title.textContent = "Dhundh Raha Hun ...";
            dlBtn.href = "#";
            dlBtn.style.pointerEvents = "none";
            dlBtn.style.opacity = "0.5";

            // Animate dice
            dice.style.animation = "none";
            void dice.offsetWidth;
            dice.style.animation = "rollDice 0.5s ease";

            try {
                const res = await fetch(CONFIG.apiUrl, {
                    method: 'POST',
                    headers: { 'Content-Type': 'application/json' },
                    body: JSON.stringify({ featured: true, seen: state.seenSurprises })
                });

                const result = await res.json();
                const novels = result.data || [];
                
                if (novels.length === 0) { 
                    title.textContent = "Koi featured novel nahi mila, dobara try karein."; 
                    return; 
                }
                
                const pick = novels[0]; 
                title.textContent = pick.Titles;
                logSurpriseEvent(isReroll ? "reroll" : "click", pick.Titles);
                
                const safeLink = (pick.Links && /^https?:\/\//i.test(pick.Links)) ? pick.Links : '#';
                dlBtn.href = safeLink;
                dlBtn.style.pointerEvents = "auto";
                dlBtn.style.opacity = "1";
                dlBtn.setAttribute("data-novel", pick.Titles);

                setTimeout(() => {
                    if (overlay.classList.contains("active") && againBtn) {
                        againBtn.classList.add('attention');
                    }
                }, 2500);

                if (result.reset) {
                    state.seenSurprises = [];
                }

                if (!state.seenSurprises.includes(pick.Titles)) {
                    state.seenSurprises.push(pick.Titles);
                }
                
                if (state.seenSurprises.length >= 5000) {
                    state.seenSurprises = [];
                    localStorage.removeItem("seenSurprises");
                    showToast("Surprise history reset ho gayi hai!");
                } else {
                    localStorage.setItem("seenSurprises", JSON.stringify(state.seenSurprises));
                }
                
            } catch(e) {
                console.error("Surprise Me Error:", e);
                title.textContent = "Error aa gaya, dobara try karein.";
            }
        }

        function startVoiceSearch() {
            const SR = window.SpeechRecognition || window.webkitSpeechRecognition;
            if (!SR) { showToast("Voice search not supported on this browser."); return; }
            const r = new SR();
            
            r.lang = 'en-US'; 
            
            r.onstart = () => { 
                isListening = true; 
                clearTimeout(typewriterTimeout); 
                
                document.getElementById("searchBox").value = ""; 
                toggleClearBtn(); 
                document.getElementById("suggestions").classList.remove("active");

                document.getElementById("searchBox").placeholder = "Listening... Speak now"; 
                document.querySelector(".mic").classList.add("listening"); 
            };
            
            r.onend = () => { 
                isListening = false; 
                document.getElementById("searchBox").placeholder = "Search novel title..."; 
                document.querySelector(".mic").classList.remove("listening"); 
                
                if (document.getElementById("searchBox").value.trim() === "" && !isFocused) {
                    charIdx = 0;
                    isDeleting = false;
                    typeWriterPlaceholder();
                }
            };

            r.onresult = (e) => { 
                const transcript = e.results[0][0].transcript; 
                document.getElementById("searchBox").value = transcript; 
                toggleClearBtn(); 
                searchNovel(); 
            };

            r.onerror = (event) => {
                if (event.error === 'no-speech' || event.error === 'audio-capture') {
                    showToast("Didn't catch that. Please try again.");
                } else if (event.error === 'not-allowed') {
                    showToast("Microphone access denied.");
                }
            };

            r.start();
        }

        function logSearch(q, n) { 
            const userId = getUserId();
            const deviceType = /Mobi|Android|iPhone/i.test(navigator.userAgent) ? "Mobile" : "Desktop";
            
            const resultStatus = n ? "Yes" : "No";

            const params = new URLSearchParams({ 
                search: q, 
                timestamp: new Date().toISOString(), 
                noResults: resultStatus,
                device: deviceType, 
                userId: userId 
            });
            
            fetch(CONFIG.googleScript, { method: 'POST', body: params }).catch(e => {}); 
        }
